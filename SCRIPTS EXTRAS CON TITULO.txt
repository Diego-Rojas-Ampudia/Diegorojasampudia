- Espesor 1000 a 500 con vientos

c
set display color white
c
run jaecol.gs
set lat -25 -5
set lon -80 -60
set t 16
set mpdset hires
set poli on
set clopts -1 -1. 0.12
set ylopts 1 5 0.14
set xlopts 1 5 0.14
set map 79 1 6
set xlint 5
set ylint 5
define z5=z(lev=500)
define z1=z(lev=1000)
define espesor=z5-z1
set grads off
set gxout shaded
set clevs 5500 5520 5530 5540 5550 5560 5570 5580 5600 5620 5630 5640 5650 5660 
set rbcols 59 58 57 49 47 45 43 42 41 21 22 23 24 25 26 27 28 29
set csmooth on
set cint 10
d espesor/10
set lev 500
set gxout barb
d skip(u,5,5);v
cbarn
draw title Analisis H+4 (10/02/2021-15UTC)
